1. Tạo key
keytool -genkey -alias tomcat -keystore /opt/apache-tomcat-8.5.41/bin/keystore -storetype PKCS12 -keyalg RSA -storepass amaster123 -validity 730 -keysize 2048

<Connector port="8443"
